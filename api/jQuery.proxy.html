                  jQuery.proxy()

                  ⚠ Deprecated 3.3method Returns: Function

                  Takes a function and returns a new one that will always have a particular context.

                  Signatures

                  1.4
                  jQuery.proxy()(function, context)
                  ParameterTypeDescription
                  functionFunctionThe function whose context will be changed.
                  contextPlainObjectThe object to which the context () of the function should be set.
                  1.4
                  jQuery.proxy()(context, name)
                  ParameterTypeDescription
                  contextPlainObjectThe object to which the context of the function should be set.
                  nameStringThe name of the function whose context will be changed (should be a property of theobject).
                  1.6
                  jQuery.proxy()(function, context, [additionalArguments])
                  ParameterTypeDescription
                  functionFunctionThe function whose context will be changed.
                  contextPlainObjectThe object to which the context () of the function should be set.
                  additionalArguments (optional)AnythingAny number of arguments to be passed to the function referenced in theargument.
                  1.6
                  jQuery.proxy()(context, name, [additionalArguments])
                  ParameterTypeDescription
                  contextPlainObjectThe object to which the context of the function should be set.
                  nameStringThe name of the function whose context will be changed (should be a property of theobject).
                  additionalArguments (optional)AnythingAny number of arguments to be passed to the function named in theargument.

                  Examples

                  Example 1: Change the context of functions bound to a click handler using the "function, context" signature. Unbind the first handler after first click.

                  HTML

                  
<p><button type="button" id="test">Test</button></p>
<div id="log"></div>

                  jQuery

                  
var me = {
  type: "zombie",
  test: function( event ) {
    // Without proxy, `this` would refer to the event target
    // use event.target to reference that element.
    var element = event.target;
    $( element ).css( "background-color", "red" );

    // With proxy, `this` refers to the me object encapsulating
    // this function.
    $( "#log" ).append( "Hello " + this.type + "<br>" );
    $( "#test" ).off( "click", this.test );
  }
};

var you = {
  type: "person",
  test: function( event ) {
    $( "#log" ).append( this.type + " " );
  }
};

// Execute you.test() in the context of the `you` object
// no matter where it is called
// i.e. the `this` keyword will refer to `you`
var youClick = $.proxy( you.test, you );

// attach click handlers to #test
$( "#test" )
  // this === "zombie"; handler unbound after first click
  .on( "click", $.proxy( me.test, me ) )

  // this === "person"
  .on( "click", youClick )

  // this === "zombie"
  .on( "click", $.proxy( you.test, me ) )

  // this === "<button> element"
  .on( "click", you.test );

                  Example 2: Enforce the context of the function using the "context, function name" signature. Unbind the handler after first click.

                  HTML

                  
  <p><button id="test">Test</button></p>
  <p id="log"></p>

                  jQuery

                  
var obj = {
  name: "John",
  test: function() {
    $( "#log" ).append( this.name );
    $( "#test" ).off( "click", obj.test );
  }
};
$( "#test" ).on( "click", jQuery.proxy( obj, "test" ) );

                  Example 3: Change the context of a function bound to the click handler,

                  HTML

                  
<p><button type="button" id="test">Test</button></p>
<div id="log"></div>

                  jQuery

                  
var me = {
  // I'm a dog
  type: "dog",

  // Note that event comes *after* one and two
  test: function( one, two, event ) {
    $( "#log" )

      // `one` maps to `you`, the 1st additional
      // argument in the $.proxy function call
      .append( "<h3>Hello " + one.type + ":</h3>" )

      // The `this` keyword refers to `me`
      // (the 2nd, context, argument of $.proxy)
      .append( "I am a " + this.type + ", " )

      // `two` maps to `they`, the 2nd additional
      // argument in the $.proxy function call
      .append( "and they are " + two.type + ".<br>" )

      // The event type is "click"
      .append( "Thanks for " + event.type + "ing." )

      // The clicked element is `event.target`,
      // and its type is "button"
      .append( "the " + event.target.type + "." );
  }
};

var you = { type: "cat" };
var they = { type: "fish" };

// Set up handler to execute me.test() in the context
// of `me`, with `you` and `they` as additional arguments
var proxy = $.proxy( me.test, me, you, they );

$( "#test" )
  .on( "click", proxy );
